#Menubar

Horizontal navigation bar with dropdown support.

#Implementation Example

import { Menubar } from 'lupine.components';

const menuItems = [
  { text: 'Home', link: '/' },
  {
    text: 'Products',
    children: [
      { text: 'LupineJS', link: '/lupine' },
      { text: 'Components', link: '/components' },
    ],
  },
];

<Menubar items={menuItems} />;
